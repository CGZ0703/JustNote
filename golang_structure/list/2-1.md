### 算法 2-1 & 算法 2-2 


```go
package list

import "fmt"

/*
	算法2-1
	将所有在线性表Lb中但不在La中的数据元素存入到La中(不改变表lb)
*/
func Union(La *List, Lb List) {
	var e interface{}

	La_len := La.ListLength()

	Lb_len := Lb.ListLength()

	for i := 1; i < Lb_len; i++ {
		e = Lb.GetElem(i)
		if La.LocateElem(e, equal) > 0 {
			La.ListInsert(i, e)
			// 实际过程中不需要这样
			La_len++
		}

	}
}

func equal() {
	fmt.Println("equal")
}

/*
	算法2-2
	已知线性表La和Lb中的数据元素按值非递减有序排列
	归并La和Lb得到新的线性表Lc，Lc的数据元素也按值非递减有序排列(不改变La和Lb)

	注: 本算法中将元素类型指定为 int

*/

func MergeList(La List, Lb List, Lc *List) {
	La_len := La.ListLength()
	Lb_len := Lb.ListLength()
	// ai,bj 分别表示 la中第i个元素，和lb中第j个元素
	var ai, bj int

	// 创建一个空的线性表
	Lc = new(List)

	// i 表示la中元素序号
	// j 表示lb中元素序号
	// k 表示lc中元素序号
	i, j, k := 1, 1, 0
	for i <= La_len && j <= Lb_len {
		ai = La.GetElem(i).(int)
		bj = Lb.GetElem(j).(int)
		if ai <= bj {
			// 在lc的最后插入元素ai
			Lc.ListInsert(k, ai)
			i++
			k++

		} else {
			Lc.ListInsert(k, bj)
			j++
			k++
		}
	}

	//下面的两个循环中只有一个会被执行

	// 表la 中还有元素未插入到lc中
	for i <= La_len {
		ai = La.GetElem(i).(int)
		Lc.ListInsert(k, ai)
		k++
		i++
	}

	// 表lb 中还有元素未插入到lc中
	for j <= Lb_len {
		bj = Lb.GetElem(j).(int)
		Lc.ListInsert(j, bj)
		k++
		j++
	}
}
```
